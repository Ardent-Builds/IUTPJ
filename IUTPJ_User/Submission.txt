#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int a,b,c;
    while(54)
    {
        cin>>a>>b>>c;
       
        if(a<b)
        {
            swap(a,b);
            if(a<c)
                swap(a,c);
        }
        else if(a<c)
            swap(a,c);

        if(a*a==(b*b)+(c*c))
            cout<<"right\n";
        else
            cout<<"wrong\n";
    }

    return 0;
}
